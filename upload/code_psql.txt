1. For the disease_id :
CREATE SEQUENCE disease_id_seq START 1;
CREATE OR REPLACE FUNCTION generate_disease_id()
RETURNS TEXT AS $$
BEGIN
  RETURN 'DIS' || LPAD(nextval('disease_id_seq')::TEXT, 5, '0');
END;
$$ LANGUAGE plpgsql;

2. For Citizen ID:

CREATE SEQUENCE citizen_seq START 1;
CREATE OR REPLACE FUNCTION generate_citizen_id()
RETURNS TEXT AS $$
DECLARE
    seq INT;
BEGIN
    seq := nextval('citizen_seq');

    RETURN 'SMC-' || EXTRACT(YEAR FROM NOW()) || '-' ||
           LPAD(seq::TEXT,6,'0');
END;
$$ LANGUAGE plpgsql;

3. For Appointment ID:
CREATE OR REPLACE FUNCTION generate_appointment_id()
RETURNS TEXT AS $$
BEGIN
RETURN 'APP' ||
TO_CHAR(NOW(),'DDMMHH24MISSMS');
END;
$$ LANGUAGE plpgsql;

4. For Health Index ID
CREATE SEQUENCE health_index_seq START 1;
CREATE OR REPLACE FUNCTION generate_index_id()
RETURNS TEXT AS $$
DECLARE
seq INT;
BEGIN
seq := nextval('health_index_seq');
RETURN 'IDX' || seq;
END;
$$ LANGUAGE plpgsql;
